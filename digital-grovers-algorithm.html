Chapter 23: Parallel Digital Implementation of Grover’s Algorithm

This chapter presents a parallel digital implementation of Grover’s algorithm. By utilizing the analytical discrete-time mapping of the propositional space [Chapter 13, Chapter 21], the amplitude state vector is propagated directly using parallel geometric reflections on digital hardware. This approach maps the state space directly and preserves the identical \(\mathcal{O}(\sqrt{N})\) complexity scaling.

23.1 Discrete Analytical Operators

Let the joint state of the \(N\) propositions be represented by the discrete amplitude vector \(\mathbf{A} = [a_1, a_2, \dots, a_N]^T \in \mathbb{R}^N\). The state transitions are calculated using two analytical reflections:

  1. Analytical Oracle Reflection (\(\mathbf{U}_w\)): The phase of the target proposition \(t\) is inverted, which is an exact \(O(1)\) parallel operation: \[a_j \to \begin{cases} -a_j & \text{if } j = t \\ a_j & \text{otherwise} \end{cases}\]
  2. Analytical Mean Reflection (\(\mathbf{U}_s\)): We compute the spatial average amplitude: \[\langle a \rangle = \frac{1}{N} \sum_{i=1}^N a_i\] Each state is reflected about this mean, which requires an \(O(\log N)\) parallel reduction: \[a_j \to 2\langle a \rangle - a_j\]

By applying these discrete operators sequentially, the target proposition’s amplitude is amplified to its maximum peak in exactly:

\[M = \left\lfloor \frac{\pi}{4} \sqrt{N} \right\rfloor \text{ steps}\]

This discrete-time parallel digital implementation operates with exact algebraic steps while preserving the identical \(\mathcal{O}(\sqrt{N})\) complexity scaling on parallel hardware.

23.2 Ada 2022 Implementation

Here is the complete, compilable Ada 2022 parallel digital implementation of Grover’s algorithm.

1. Package Specification (parallel_grover.ads)

generic
   type Real is digits <>;
package Parallel_Grover is

   type Amplitude_Array is array (Positive range <>) of Real;

   -- Initializes the proposition amplitudes to the uniform state: 1/sqrt(N)
   procedure Initialize (Amplitudes : out Amplitude_Array) with
     Pre => Amplitudes'Length > 0 and then Amplitudes'First = 1;

   -- Executes one exact discrete Grover iteration step in parallel
   procedure Discrete_Grover_Step (
      Amplitudes : in out Amplitude_Array;
      Target     : in Positive
   ) with
     Pre => Amplitudes'Length > 0 
            and then Amplitudes'First = 1 
            and then Target in Amplitudes'Range;

end Parallel_Grover;

2. Package Body (parallel_grover.adb)

with Ada.Numerics.Generic_Elementary_Functions;

package body Parallel_Grover is

   package Math is new Ada.Numerics.Generic_Elementary_Functions (Real);
   use Math;

   -- Parallel divide-and-conquer binary sum reduction: O(log N)
   function Parallel_Sum (Amps : Amplitude_Array; Low, High : Positive) return Real is
      Threshold : constant Positive := 16; -- Falls back to serial sum on small slices
   begin
      if High - Low <= Threshold then
         declare
            Sum_Val : Real := 0.0;
         begin
            for I in Low .. High loop
               Sum_Val := Sum_Val + Amps (I);
            end loop;
            return Sum_Val;
         end;
      else
         declare
            Mid : constant Positive := Low + (High - Low) / 2;

            task type Left_Worker is
               entry Get_Result (Val : out Real);
            end Left_Worker;

            task body Left_Worker is
               Local_Val : Real;
            begin
               Local_Val := Parallel_Sum (Amps, Low, Mid);
               accept Get_Result (Val : out Real) do
                  Val := Local_Val;
               end Get_Result;
            end Left_Worker;

            Left_Task : Left_Worker;
            Right_Sum : Real;
            Left_Sum  : Real;
         begin
            Right_Sum := Parallel_Sum (Amps, Mid + 1, High);
            Left_Task.Get_Result (Left_Sum);
            return Left_Sum + Right_Sum;
         end;
      end if;
   end Parallel_Sum;

   ----------------
   -- Initialize --
   ----------------
   procedure Initialize (Amplitudes : out Amplitude_Array) is
      N_Factor : constant Real := 1.0 / Sqrt (Real (Amplitudes'Length));
   begin
      parallel for I in Amplitudes'Range loop
         Amplitudes (I) := N_Factor;
      end loop;
   end Initialize;

   --------------------------
   -- Discrete_Grover_Step --
   --------------------------
   procedure Discrete_Grover_Step (
      Amplitudes : in out Amplitude_Array;
      Target     : in Positive
   ) is
      Temp_Amps : Amplitude_Array (Amplitudes'Range);
      Mean_Amp  : Real;
   begin
      -- 1. Parallel Oracle Phase Reflection: O(1)
      parallel for I in Amplitudes'Range loop
         if I = Target then
            Temp_Amps (I) := -Amplitudes (I);
         else
            Temp_Amps (I) := Amplitudes (I);
         end if;
      end loop;

      -- 2. Parallel Mean Computation: O(log N)
      Mean_Amp := Parallel_Sum (Temp_Amps, Temp_Amps'First, Temp_Amps'Last) / Real (Temp_Amps'Length);

      -- 3. Parallel Reflection-about-the-mean: O(1)
      parallel for I in Amplitudes'Range loop
         Amplitudes (I) := (2.0 * Mean_Amp) - Temp_Amps (I);
      end loop;
   end Discrete_Grover_Step;

end Parallel_Grover;

3. Main Driver Program (run_parallel_grover.adb)

with Parallel_Grover;
with Ada.Text_IO;
with Ada.Real_Time;
with Ada.Numerics.Generic_Elementary_Functions;

procedure Run_Parallel_Grover is
   use Ada.Real_Time;

   type Precision_Real is digits 15;
   package Real_IO is new Ada.Text_IO.Float_IO (Precision_Real);
   package Int_IO is new Ada.Text_IO.Integer_IO (Integer);
   package Math is new Ada.Numerics.Generic_Elementary_Functions (Precision_Real);

   package Solver is new Parallel_Grover (Precision_Real);
   use Solver;

   -- Database configuration
   N              : constant Positive := 1024; -- Database scale (1024 propositions)
   Target_Index   : constant Positive := 712;  -- Searching for index 712
   Amplitudes     : Amplitude_Array (1 .. N);
   
   -- Analytical iteration scale: floor(pi/4 * sqrt(N))
   Num_Iterations : constant Positive := Positive (
      Precision_Real'Floor (
         (Ada.Numerics.Pi / 4.0) * Math.Sqrt (Precision_Real (N))
      )
   );

   Start_Time, End_Time : Time;
   Elapsed              : Time_Span;

begin
   Ada.Text_IO.Put_Line ("Starting Parallel Digital Implementation of Grover's Algorithm...");
   Ada.Text_IO.Put_Line ("Propositions (N):   " & Positive'Image (N));
   Ada.Text_IO.Put_Line ("Target Index:       " & Positive'Image (Target_Index));
   Ada.Text_IO.Put_Line ("Target Iterations:  " & Positive'Image (Num_Iterations));
   Ada.Text_IO.New_Line;

   Start_Time := Clock;

   -- Initialize uniform superposition of logical propositions
   Initialize (Amplitudes);

   -- Execute discrete-time parallel updates
   for Iter in 1 .. Num_Iterations loop
      Discrete_Grover_Step (Amplitudes, Target_Index);
   end loop;

   End_Time := Clock;
   Elapsed  := End_Time - Start_Time;

   -- Output performance metrics
   Ada.Text_IO.Put_Line ("Search complete.");
   Ada.Text_IO.Put ("Elapsed Wall-Clock Computation Time: ");
   Real_IO.Put (Precision_Real (To_Duration (Elapsed)), Fore => 1, Aft => 6, Exp => 0);
   Ada.Text_IO.Put_Line (" seconds.");
   Ada.Text_IO.New_Line;

   -- Output results
   Ada.Text_IO.Put_Line ("Final Proposition Amplitudes:");
   Ada.Text_IO.Put_Line ("Index   |  Amplitude  | Probability (Amplitude^2)");
   Ada.Text_IO.Put_Line ("--------+-------------+---------------------------");
   for I in Amplitudes'Range loop
      -- Output only the bounds and target area to keep console readable
      if I = Target_Index or I <= 3 or I >= N - 2 then
         Int_IO.Put (I, Width => 5);
         Ada.Text_IO.Put ("   | ");
         Real_IO.Put (Amplitudes (I), Fore => 3, Aft => 8, Exp => 0);
         Ada.Text_IO.Put ("  | ");
         Real_IO.Put (Amplitudes (I) ** 2, Fore => 3, Aft => 8, Exp => 0);
         if I = Target_Index then
            Ada.Text_IO.Put ("  <-- Target Identified");
         end if;
         Ada.Text_IO.New_Line;
      elsif I = Target_Index - 1 or I = Target_Index + 1 then
         Ada.Text_IO.Put_Line ("  ...   |     ...     |            ...");
      end if;
   end loop;

end Run_Parallel_Grover;
